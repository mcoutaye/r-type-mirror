68void Client::update()
69{
70 _timer.updateClock();
71 std::vector<EntityUpdate> updates;
72
73 while (_UDP.receivedUpdates.tryPop(updates)) {
74 for (auto &update : updates)
75 this->applyUpdate(update);
76 }
77 _moveSystem.update(1.0f / 60.f);
78
79 // Remove entities out of bounds
80 auto entities = _ecs.getEntitiesByComponents<Position_t>();
81 for (Entity e : entities) {
82 // Don't kill the local player if they go out of bounds (let server handle it or block movement)
83 if (e == _localPlayerEntity) continue;
84
85 auto* pos = _ecs.getComponent<Position_t>(e);
86 if (pos && (pos->x < -100.f || pos->x > 2200.f || pos->y < -100.f || pos->y > 1200.f)) {
87 _ecs.killEntity(e);
88 }
89 }
90
91 // Client side collision to remove projectiles
92 auto projectiles = _ecs.getEntitiesByComponents<Projectile_t, Position_t, Collider_t>();
93 auto targets = _ecs.getEntitiesByComponents<Position_t, Collider_t>();
94
95 for (Entity p : projectiles) {
96 auto projPos = _ecs.getComponent<Position_t>(p);
97 auto projCol = _ecs.getComponent<Collider_t>(p);
98
99 if (!projPos || !projCol) continue;
100
101 for (Entity t : targets) {
102 if (p == t) continue;
103 if (t == _localPlayerEntity) continue;
104 if (_ecs.hasComponent<Projectile_t>(t)) continue;
105
106 auto targetPos = _ecs.getComponent<Position_t>(t);
107 auto targetCol = _ecs.getComponent<Collider_t>(t);
108
109 if (!targetPos || !targetCol) continue;
110
111 if (projPos->x < targetPos->x + targetCol->width &&
112 projPos->x + projCol->width > targetPos->x &&
113 projPos->y < targetPos->y + targetCol->height &&
114 projPos->y + projCol->height > targetPos->y) {
115
116 _ecs.killEntity(p);
117 break;
118 }
119 }
120 }
121}

123void Client::applyUpdate(EntityUpdate &update)
124{
125 Entity entity = 0;
126
127 if (serverToClientEntityRelation.find(update.entityId) == serverToClientEntityRelation.end()) {
128 // Entity does not exist for client, create it
129 entity = _ecs.createEntity();
130 serverToClientEntityRelation[update.entityId] = entity;
131
132 // Suppose entity is Player, but can be everything else.
133 // should add EntityType to EntityUpdate later.
134 _ecs.addComponents<Position_t, Health_t, Drawable_t, Collider_t>(entity,
135 {update.position.x, update.position.y},
136 {update.health.current, update.health.max},
137 {"ship.png", sf::IntRect(0, 0, 64, 64), 10, true, 1.f, 0.f},
138 Collider_t{64.f, 64.f, true, 2, 0});
139
140 } else {
141 entity = serverToClientEntityRelation[update.entityId];
142 }
143
144 auto pos = _ecs.getComponent<Position_t>(entity);
145 auto health = _ecs.getComponent<Health_t>(entity);
146
147 if (pos) {
148 pos->x = update.position.x;
149 pos->y = update.position.y;
150 }
151 if (health) {
152 health->current = update.health.current;
153 health->max = update.health.max;
154 }
155
156 if (update.tick == 0xFFFF) {
157 _localPlayerEntity = entity;
158 } else if (update.tick == 0xFFFE || update.tick == 0xFFFD) {
159 // Entity died
160 _ecs.killEntity(entity);
161 if (serverToClientEntityRelation.find(update.entityId) != serverToClientEntityRelation.end()) {
162 serverToClientEntityRelation.erase(update.entityId);
163 }
164 if (update.tick == 0xFFFD) {
165 // Played killed it, maybe play sound or add score
166 std::cout << "I killed entity " << update.entityId << "!" << std::endl;
167 }
168 }
169}

176void Client::processInput()
177{
178 InputState inputs = {0, 0, 0, 0, 0, 0};
179 inputs.tick = _timer.getCurrentFrame();
180 _inputSystem.update(0);
181 if (_inputSystem.isActionActive(GameAction::MoveUp))
182 inputs.up = 1;
183 if (_inputSystem.isActionActive(GameAction::MoveDown))
184 inputs.down = 1;
185 if (_inputSystem.isActionActive(GameAction::MoveLeft))
186 inputs.left = 1;
187 if (_inputSystem.isActionActive(GameAction::MoveRight))
188 inputs.right = 1;
189 if (_inputSystem.isActionActive(GameAction::Shoot)) {
190 inputs.shoot = 1;
191 if (_shootCooldown <= 0.f) {
192 if (_localPlayerEntity != -1) {
193 auto pos = _ecs.getComponent<Position_t>(_localPlayerEntity);
194 if (pos) {
195 Factory::createProjectile(_ecs, pos->x + 64.f, pos->y + 20.f, 800.f, 0.f, 1, 25, "bullet");
196 _shootCooldown = SHOOT_DELAY;
197 }
198 }
199 }
200 }
201 if (_shootCooldown > 0.f)
202 _shootCooldown -= 1.0f / 60.f;
203
204 if (_inputSystem.isActionActive(GameAction::Quit))
205 _running = false;
206 _UDP.inputsToSend.push(inputs);
207}
