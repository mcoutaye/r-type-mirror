115 void add(Entity e, const ComponentType &component)

168 {
169 assert(e < MAX_ENTITIES);
170 __checker.reset(e);

295 template<typename ComponentType>
296 bool hasComponent(Entity e) const
297 {
298 assert(e < MAX_ENTITIES);
299 auto TypeID = getComponentID<ComponentType>();
300 // Signatures are telling if entity SHOULD have a component or not
301 // It MAY be possible that an entity has a component set, but signature denies it.
302 // Always trust Signature, All composents will be destroyed proprely at ~ECS
303 auto it = __signatures.find(e);
304 if (it == __signatures.end())
305 return false;
306 if (!it->second.test(TypeID))
307 return false;
308 // Signature say entity SHOULD have this, for safety we do internal checks
309 if (TypeID >= _componentStorages.size())
310 return false;
311 if (_componentStorages[TypeID] == nullptr)
312 return false;
313 auto *storage = static_cast<StorageComponent<ComponentType> *>(_componentStorages[TypeID]);
314 return storage->has(e);
315 }
