49template<typename ComponentType>
50class StorageComponent
51{
52 public:
53 StorageComponent()
54 {
55 __Storage.resize(MAX_ENTITIES); // Posibly one for every entity
56 __checker.reset();
57 }
58 ~StorageComponent() = default;
59
60 // FUCK storage : Find, Update, Create & Kill for component
61 void add(Entity e, const ComponentType &component)
62 {
63 assert(e < MAX_ENTITIES);
64
65 __Storage[e] = component; //component for 'e' entity is in [e]
66 __checker.set(e); // Update checker
67 }
68
69 bool has(Entity e) const
70 {
71 assert(e < MAX_ENTITIES);
72
73 return __checker.test(e); // Check 'e' bit to see if 'e' entity has this component type
74 }
75
76 const ComponentType *get(Entity e) const // READ-ONLY version because const (FIND)
77 {
78 assert(e < MAX_ENTITIES);
79 if (!has(e))
80 return nullptr;
81
82 return &__Storage[e]; // Reference to the component to modify it IN SYSTEM ONLY
83 }
84
85 ComponentType *get(Entity e) // get version for UPDATE
86 {
87 assert(e < MAX_ENTITIES);
88 if (!has(e))
89 return nullptr;
90
91 return &__Storage[e];
92 }
93
94 void kill(Entity e)
95 {
96 assert(e < MAX_ENTITIES);
97
98 __checker.reset(e);
99 // No need to remove in __Storage, the [e] space is for this entity anyways.
100 // In case of removing/re-adding, its better since we just change one bit instead of playing with memory
101 }
102
103 private:
104 // Where the components are stocked.
105 std::vector<ComponentType> __Storage;
106
107 // Those bits are check to know if an entity has a component of this specific type
108 // It doesnt tell which one, just if it has this type or not.
109 std::bitset<MAX_ENTITIES> __checker;
110};

169 Entity addComponent(Entity e, const ComponentType &component)
170 {
171 auto typeID = getComponentID<ComponentType>(); // Fetch the TypeID of this ComponentType
172 verifyComponentStorage<ComponentType>(typeID);
173
174 auto *storage = static_cast<StorageComponent<ComponentType> *>(__componentStorage[typeID]); // Get the storage of the componant type
175 storage->add(e, component); // Add in storage
176 __signatures[e].set(typeID); // Set signature for the new added component
177

197 template<typename ComponentType>
198 bool hasComponent(Entity e) const
199 {
200 assert(e < MAX_ENTITIES);
201
202 auto TypeID = getComponentID<ComponentType>();
203
204 // Signatures are telling if entity SHOULD have a component or not
205 // It MAY be possible that an entity has a component set, but signature denies it.
206 // Always trust Signature, All composents will be destroyed proprely at ~ECS
207 auto it = __signatures.find(e);
208 if (it == __signatures.end())
209 return false;
210 if (!it->second.test(TypeID))
211 return false;
212
213 // Signature say entity SHOULD have this, for safety we do internal checks
214 if (TypeID >= __componentStorage.size())
215 return false;
216 if (__componentStorage[TypeID] == nullptr)
217 return false;
218
219 auto *storage = static_cast<StorageComponent<ComponentType> *>(__componentStorage[TypeID]);
220 return storage->has(e);
221 }

264 {
