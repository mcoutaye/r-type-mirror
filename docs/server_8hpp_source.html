80void Server::applyInput(std::pair<int, InputState> &currentItem)
81{
82 if (clientToPlayerRelation.find(currentItem.first) == clientToPlayerRelation.end())
83 return;
84
85 Entity e = clientToPlayerRelation[currentItem.first];
86 auto velo = _ecs.getComponent<Velocity_t>(e);
87
88 if (!velo)
89 return;
90
91 float speed = 400.0f;
92 velo->x = 0;
93 velo->y = 0;
94
95 if (currentItem.second.right)
96 velo->x = speed;
97 if (currentItem.second.left)
98 velo->x = -speed;
99 if (currentItem.second.up)
100 velo->y = -speed;
101 if (currentItem.second.down)
102 velo->y = speed;
103
104 auto ctrl = _ecs.getComponent<PlayerController_t>(e);
105 if (ctrl) {
106 ctrl->isShooting = currentItem.second.shoot;
107 }
108}

110void Server::processInputs()
111{
112 // Reset velocity for all players to prevent sliding
113 for (auto& pair : clientToPlayerRelation) {
114 auto velo = _ecs.getComponent<Velocity_t>(pair.second);
115 if (velo) {
116 velo->x = 0;
117 velo->y = 0;
118 }
119 }
120
121 this->entityForClients();
122 std::pair<int, InputState> first;
123 std::pair<int, InputState> input;
124
125 if (!_UDP.receivedInputs.tryPop(first))
126 return; // No input
127 input = first;
128
129 while (first.second.tick <= input.second.tick) {
130 this->applyInput(input);
131 // Do the next arg
132 if (!_UDP.receivedInputs.tryPop(input))
133 return; // No more inputs
134 }
135
136 // still process the last input, you cant put it back in the queue anyways
137 this->applyInput(input);
138}

140void Server::update()
141{
142 _movSys.update(1.0f / 60.f); // Update with fixed delta time
143 _movementSystem.update(1.0f / 60.f);
144 _waveSystem.update(1.0f / 60.f);
145
146 // Remove entities out of bounds
147 auto entities = _ecs.getEntitiesByComponents<Position_t>();
148 for (Entity e : entities) {
149 if (_ecs.hasComponent<PlayerController_t>(e))
150 continue;
151
152 auto* pos = _ecs.getComponent<Position_t>(e);
153 if (pos->x < -100.f || pos->x > 2200.f || pos->y < -100.f || pos->y > 1200.f) {
154 _ecs.killEntity(e);
155 }
156 }
157
158 for (Entity e : _ecs.getEntitiesByComponents<PlayerController_t, Position_t>()) {
159 auto* ctrl = _ecs.getComponent<PlayerController_t>(e);
160 auto* pos = _ecs.getComponent<Position_t>(e);
161
162 if (ctrl && ctrl->isShooting && ctrl->shootCooldown <= 0.f) {
163 Factory::createProjectile(_ecs, pos->x + 64.f, pos->y + 20.f, 800.f, 0.f, 1, 50, "bullet", e);
164 ctrl->shootCooldown = SHOOT_DELAY;
165 }
166 ctrl->shootCooldown -= 1.0f / 60.f;
167 }
168
169 // Collision detection
170 auto projectiles = _ecs.getEntitiesByComponents<Projectile_t, Collider_t, Position_t>();
171 auto targets = _ecs.getEntitiesByComponents<Health_t, Collider_t, Position_t>();
172
173 for (Entity p : projectiles) {
174 auto proj = _ecs.getComponent<Projectile_t>(p);
175 auto projCol = _ecs.getComponent<Collider_t>(p);
176 auto projPos = _ecs.getComponent<Position_t>(p);
177
178 for (Entity t : targets) {
179 if (p == t) continue;
180 auto targetCol = _ecs.getComponent<Collider_t>(t);
181 auto targetPos = _ecs.getComponent<Position_t>(t);
182 auto targetHealth = _ecs.getComponent<Health_t>(t);
183
184 // Check teams (1=player, 2=enemy)
185 // if (projCol->team == targetCol->team) continue;
186 if (projCol->team == targetCol->team && targetCol->team != 0) continue;
187
188 // AABB Collision
189 if (projPos->x < targetPos->x + targetCol->width &&
190 projPos->x + projCol->width > targetPos->x &&
191 projPos->y < targetPos->y + targetCol->height &&
192 projPos->y + projCol->height > targetPos->y) {
193
194 // Hit!
195 targetHealth->current -= proj->damage;
196 targetHealth->lastAttackerId = proj->ownerId;
197 _ecs.killEntity(p); // Destroy projectile
198 break; // Projectile destroyed, move to next projectile
199 }
200 }
201 }
202 for (Entity e : _ecs.getEntitiesByComponents<Projectile_t, Position_t>()) {
203 auto* pos = _ecs.getComponent<Position_t>(e);
204 if (pos->x > 2000.f || pos->x < -100.f || pos->y > 1200.f || pos->y < -100.f)
205 _ecs.killEntity(e);
206 }
207}

209void Server::broadcast()
210{
211 // send every player entities or projectile for the moment
212 auto entities = _ecs.getEntitiesByComponents<Position_t, Health_t, SendUpdate_t>(); // Players & Enemies
213 auto clients = _UDP.getClients();
214 uint16_t serverTick = _timer.getCurrentFrame();
215
216 for (Entity e : entities) {
217 auto pos = _ecs.getConstComponent<Position_t>(e);
218 auto pv = _ecs.getConstComponent<Health_t>(e);
219
220 // Check if this entity belongs to a client
221 int ownerClientId = -1;
222 for (const auto& pair : clientToPlayerRelation) {
223 if (pair.second == e) {
224 ownerClientId = pair.first;
225 break;
226 }
227 }
228
229 for (const auto& client : clients) {
230 uint16_t tickToSend = serverTick;
231 if (client.playerId == ownerClientId) {
232 tickToSend = 0xFFFF; // Magic value for "This is you"
233 }
234
235 // Handle death events
236 if (pv->current <= 0) {
237 // Check if this client killed the entity
238 // We need to find the player ID associated with the lastAttackerId entity
239 int attackerClientId = -1;
240 for (const auto& pair : clientToPlayerRelation) {
241 if (pair.second == pv->lastAttackerId) {
242 attackerClientId = pair.first;
243 break;
244 }
245 }
246
247 if (client.playerId == attackerClientId) {
248 tickToSend = 0xFFFD; // You killed it
249 } else {
250 tickToSend = 0xFFFE; // It died
251 }
252 }
253
254 // total size needed (Entity ID + Position + Health (current + max))
255 // Note: We manually serialize health to exclude lastAttackerId and match EntityUpdate struct
256 size_t totalSize = sizeof(Entity) + sizeof(uint16_t) + sizeof(Position_t) + (2 * sizeof(int));
257 std::vector<uint8_t> packetData(totalSize);
258 size_t offset = 0;
259
260 std::memcpy(packetData.data() + offset, &e, sizeof(Entity));
261 offset += sizeof(Entity);
262 std::memcpy(packetData.data() + offset, &tickToSend, sizeof(uint16_t));
263 offset += sizeof(uint16_t);
264 std::memcpy(packetData.data() + offset, pos, sizeof(Position_t));
265 offset += sizeof(Position_t);
266
267 // Manual Health serialization
268 int currentHealth = pv->current;
269 int maxHealth = pv->max;
270 std::memcpy(packetData.data() + offset, &currentHealth, sizeof(int));
271 offset += sizeof(int);
272 std::memcpy(packetData.data() + offset, &maxHealth, sizeof(int));
273
274 _UDP.packetsToSend.push({client.address, client.port, packetData});
275 }
276
277 if (pv->current <= 0) {
278 std::cout << "Entity " << e << " died (Killer: " << pv->lastAttackerId << ")" << std::endl;
279 }
280 }
281
282 // Cleanup dead entities after broadcast
283 for (Entity e : entities) {
284 auto pv = _ecs.getConstComponent<Health_t>(e);
285 if (pv->current <= 0) {
286 _ecs.killEntity(e);
287 }
288 }
289}

305void Server::entityForClients()
306{
307 std::size_t clientCount = _UDP.getClientCount();
308
309 if (clientCount == this->_nbPlayer)
310 return;
311
312 std::vector<ClientInfo> Clients = _UDP.getClients();
313
314 for (const auto &client : Clients) {
315 if (clientToPlayerRelation.find(client.playerId) == clientToPlayerRelation.end()) {
316 Entity newEntity = Factory::createPlayer(_ecs, 0.f, 0.f, client.playerId, "player");
317 clientToPlayerRelation.insert({ client.playerId, newEntity });
318 this->_nbPlayer++;
319 }
320 }
321}
